program complete_test;
var x : int;
    y : float;
    name : string;

void calculate (a : int, b : int)
[  var result : int;

   {  result = a * b + 10;
      print("Result is: ", result);
   }
];

void process (val : float)
[  var temp : float;

   {  temp = val / 2.0;
      if ( temp > 5.0 )
      {
         print("Large value");
      }
      else
      {
         print("Small value");
      };
   }
];

main
{  x = 100;
   y = 25.5;
   name = "TestProgram";

   print("Starting program: ", name);

   calculate(x, 5);

   process(y);

   do {
      x = x - 10;
      print(x);
   }while( x > 0 );

   if ( y >= 20.0 )
   {
      print("Y is at least 20");
      y = y * 1.5;
   }
   else
   {
      print("Y is less than 20");
      y = y * 2.0;
   };

   print("Final values - X:", x, "Y:", y);
}
end
