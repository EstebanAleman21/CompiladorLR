program test_relations;
var x : int;
    y : float;
    result : int;

main
{  x = 10;
   y = 5.5;

   if ( x > 5 )
   {
      print("x is greater than 5");
   };

   if ( x < 20 )
   {
      print("x is less than 20");
   };

   if ( x >= 10 )
   {
      print("x is greater than or equal to 10");
   };

   if ( y <= 10.0 )
   {
      print("y is less than or equal to 10");
   };

   if ( x == 10 )
   {
      print("x equals 10");
   };

   if ( x != 5 )
   {
      print("x is not equal to 5");
   };

   result = 1;
}
end
