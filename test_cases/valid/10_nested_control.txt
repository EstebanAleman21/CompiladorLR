program testNestedControl;
var i, j : int;

main {
  print("=== While anidado ===");

  i = 0;
  while (i < 3) {
    j = 0;
    while (j < 2) {
      print("i=", i, " j=", j);
      j = j + 1;
    };
    i = i + 1;
  };

  print("=== If anidados ===");

  i = 5;
  if (i > 0) {
    if (i > 3) {
      print("i > 3");
    } else {
      print("0 < i <= 3");
    };
  } else {
    print("i <= 0");
  };

  print("=== While con if adentro ===");

  i = 0;
  while (i < 4) {
    if (i == 2) {
      print("i es 2!");
    } else {
      print("i no es 2, es ", i);
    };
    i = i + 1;
  };

  print("=== If con while adentro ===");

  i = 1;
  if (i > 0) {
    j = 0;
    while (j < 3) {
      print("Dentro del if-while: j=", j);
      j = j + 1;
    };
  };

  print("=== Fin de pruebas ===");
}
end
